import Foundation


// Define a class for Student

class Student {

    let name: String

    var grades: [Int]


    init(name: String, grades: [Int]) {

        self.name = name

        self.grades = grades

    }


    func averageGrade() -> Double {

        let total = grades.reduce(0, +) // Sum all grades

        return grades.isEmpty ? 0.0 : Double(total) / Double(grades.count)

    }

}
